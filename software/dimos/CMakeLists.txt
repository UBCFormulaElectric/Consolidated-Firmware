# define sources
file(GLOB_RECURSE UI_SOURCES
        "${CMAKE_CURRENT_SOURCE_DIR}/src/main.cpp"

    set(DEV_SOURCES ${UI_SOURCES} ${DEV_IO_SOURCES} ${IO_INC} ${SHARED_FILES})

    message(FATAL_ERROR "‚ùå Unknown target ${TARGET}")
